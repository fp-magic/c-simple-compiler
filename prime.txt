// 输入数字num，判断是不是素数
// 测试基本的分支，循环结构
int isPrime(int x)
{
    int i;
    int res;
    i = 2;
    if(x < 2)
    {
        res = 0;
    }
    else
    {
        res = 1;
    }
    while(i * i <= x)
    {
        if(x / i * i == x)
        {
            res = 0;
        }
        i = i + 1;
    }
    return res;
}
